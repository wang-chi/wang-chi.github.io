tourLength <- function(tour, distMatrix) {
  tour <- c(tour, tour[1])
  route <- embed(tour, 2)[,2:1]
  sum(distMatrix[route])
}

getAdj <- function(tour) {
  n <- length(tour)
  from <- tour[1:(n - 1)]
  to <- tour[2:n]
  m <- n - 1
  A <- matrix(0, m, m)
  A[cbind(from, to)] <- 1
  A <- A + t(A)
  return(A)
}

# 2-d coordinates
mds <- cmdscale(city)
x <- mds[, 1]
y <- -mds[, 2]
n <- length(x)

B <- 100
fitnessMat <- matrix(0, B, 2)
A <- matrix(0, n, n)
for (b in seq(1, B)) {
  # run a GA algorithm
  GA.rep <- ga(type = "permutation", 
               fitness = tspfitness, 
               distMatrix = as.matrix(city), 
               lower = 1, 
               upper = 5,
               popSize = 10, 
               maxiter = 50, 
               run = 100, 
               pmutation = 0.2, 
               monitor = NULL)

  tour <- GA.rep@solution[1, ]
  tour <- c(tour, tour[1])
  #fitnessMat[b, 1] <- GA.rep@bestSol[GA.rep@iter]
  #fitnessMat[b, 2] <- GA.rep@mean[GA.rep@iter]
  A <- A + getAdj(tour)
}

plot.tour <- function(x, y, A) {
  n <- nrow(A)
  for (ii in seq(2, n)) {
    for (jj in seq(1, ii)) {
      w <- A[ii, jj]
      if (w > 0) 
        lines(x[c(ii, jj)], y[c(ii, jj)], lwd = w, col = "lightgray")
    }
  }
}
